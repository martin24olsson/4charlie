function displayBlogPosts(postsText) { var blogPostsContainer = document.getElementById('blogPosts'); blogPostsContainer.innerHTML = ''; var postsArray = postsText.split('\n\n'); postsArray.forEach(function(post, index) { // Split post into title and body var lines = post.split('\n'); var postTitle = lines[0]; var postBody = lines.slice(1).join('\n'); var postDiv = document.createElement('div'); postDiv.className = 'article'; var postTitleElement = document.createElement('h2'); postTitleElement.textContent = postTitle; postDiv.appendChild(postTitleElement); var postContent = document.createElement('p'); postContent.textContent = postBody; postDiv.appendChild(postContent); // Handle subtitles and blank spaces within the post lines.slice(1).forEach(function(line) { if (line.startsWith('h1')) { var subtitleElement = document.createElement('h3'); subtitleElement.textContent = line.substring(2); // Exclude the 'h1' prefix postDiv.appendChild(subtitleElement); } else if (line.startsWith('b1')) { var blankSpaceElement = document.createElement('div'); blankSpaceElement.className = 'blank-space'; postDiv.appendChild(blankSpaceElement); } }); postTitleElement.addEventListener('click', function() { // Toggle visibility of post content when the title is clicked postContent.style.display = postContent.style.display === 'none' ? 'block' : 'none'; }); blogPostsContainer.appendChild(postDiv); }); }
